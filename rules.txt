#
# h2Polar rules
#
# ACTION|METHOD|URL|ARGS ...
# 
# ! any path
# * any domain
# . wildcard domain
#
# EXAMPLES
# modify body response content
#                               |search|replace string
INJECT|GET|https://ifconfig.io/!|<html>|<html>h2Polar
#
# capture login credentials of github
CAPTURE|POST|https://github.com/session
#
# screenshot
SCREENSHOT|GET|https://*.google.com/
#
# remove "Date" header response
#                                                   |header name
REMOVE_HEADER_RESPONSE|GET|https://www.ifconfig.io/!|Date
#
# change tls ext ssl hostname (bypass asn firewalls)
#                                                   |replace tls hostname
FAKE_TLS_EXT_HOSTNAME|CONNECT|https://.youtube.com/!|www.google.com
#
# redirect one site to another
#                                |target domain, port 
REDIRECT|GET|http://ifconfig.io/!|localhost|80
#
# download body response content
#                                              |content type
DOWNLOAD_CONTENT|GET|https://.googlevideo.com/!|audio/webm
#
# remove "Accept-Encoding" header request
#                                                     header name
REMOVE_HEADER_REQUEST|GET|https://*.stackoverflow.com/|Accept-Encoding
#
# reject
REJECT|CONNECT|https://.facebook.com/!
